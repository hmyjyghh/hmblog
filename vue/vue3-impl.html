    Vue3.0实现原理

    # Vue3.0实现原理

    # 1. vue3 如何实现一个ref

    在 Vue 3 中，ref 是一个非常重要的响应式 API，用于创建一个可变的响应式对象。

    实现思路

    • 创建一个对象：该对象需要有一个 value 属性，用于存储传入的值。
    • 实现响应式：使用 Proxy 来拦截 value 属性的读取和设置操作，从而实现响应式更新。当 value 发生变化时，通知所有依赖该ref的地方进行更新。
    // 用于存储依赖的 WeakMap
const targetMap = new WeakMap()

// 收集依赖
function track(target, key) {
	let depsMap = targetMap.get(target)
	if (!depsMap) {
		targetMap.set(target, (depsMap = new Map()))
	}
	let dep = depsMap.get(key)
	if (!dep) {
		depsMap.set(key, (dep = new Set()))
	}
	// 这里简单假设当前有一个全局的 activeEffect 变量存储当前正在执行的副作用函数
	if (activeEffect) {
		dep.add(activeEffect)
	}
}

// 触发依赖更新
function trigger(target, key) {
	const depsMap = targetMap.get(target)
	if (!depsMap) {
		return
	}
	const dep = depsMap.get(key)
	if (dep) {
		dep.forEach(effect => effect())
	}
}

// 全局变量，用于存储当前正在执行的副作用函数
let activeEffect

// 定义副作用函数
function effect(fn) {
	activeEffect = fn
	fn()
	activeEffect = null
}

// 实现自定义的`ref `函数
function myRef(initialValue) {
	const refObject = {
		get value() {
			// 收集依赖
			track(refObject, 'value')
			return initialValue
		},
		set value(newValue) {
			if (initialValue !== newValue) {
				initialValue = newValue
				// 触发依赖更新
				trigger(refObject, 'value')
			}
		}
	}
	return refObject
}

// 通过 new Proxy  实现一个 ref 函数
function myRef(initialValue) {
	const refObject = {
		value: initialValue
	}

	return new Proxy(refObject, {
		get(target, key) {
			if (key === 'value') {
				// 收集依赖
				track(target, key)
			}
			return target[key]
		},
		set(target, key, newValue) {
			if (key === 'value' && target[key] !== newValue) {
				target[key] = newValue
				// 触发依赖更新
				trigger(target, key)
			}
			return true
		}
	})
}

// 使用示例
const count = myRef(0)

// 定义一个副作用函数，当 count 的值发生变化时，会重新执行
effect(() => {
	console.log('count 的值是:', count.value)
})

// 修改 count 的值
count.value = 1 // 此时会触发副作用函数，控制台会输出新的值

    代码解释

    1. targetMap：这是一个 WeakMap，用于存储每个对象的依赖关系。每个对象对应一个 Map，该 Map 中的每个键对应一个 Set，Set 中存储的是依赖该属性的副作用函数。
    2. track 函数：用于收集依赖。当读取ref的 value 属性时，会调用该函数，将当前正在执行的副作用函数添加到依赖集合中。
    3. trigger 函数：用于触发依赖更新。当ref的 value 属性发生变化时，会调用该函数，遍历依赖集合，执行其中的所有副作用函数。
    4. effect 函数：用于定义副作用函数。在执行副作用函数之前，会将其赋值给全局变量 activeEffect，以便在 track 函数中使用。
    5. myRef 函数：实现了自定义的ref功能。它返回一个对象，该对象有一个 value 属性，通过 getter 和 setter 方法实现了依赖收集和触发更新的功能。

    # 2. vue3 如何实现一个computed

    在 Vue 3 里，computed 是一个很实用的 API，它能根据响应式数据动态计算出一个新的值，并且这个计算结果是响应式的，当依赖的数据发生变化时会自动重新计算。

    实现思路

    • 依赖追踪：要实现依赖追踪，这样就能知道 computed 依赖了哪些响应式数据，当这些依赖数据发生变化时触发重新计算。
    • 缓存机制：为了避免不必要的重复计算，需要实现缓存机制，只有在依赖数据发生变化时才重新计算。
    // 用于存储依赖关系的 WeakMap
const targetMap = new WeakMap()

// 收集依赖
function track(target, key) {
	let depsMap = targetMap.get(target)
	if (!depsMap) {
		targetMap.set(target, (depsMap = new Map()))
	}
	let dep = depsMap.get(key)
	if (!dep) {
		depsMap.set(key, (dep = new Set()))
	}
	if (activeEffect) {
		dep.add(activeEffect)
	}
}

// 触发依赖更新
function trigger(target, key) {
	const depsMap = targetMap.get(target)
	if (!depsMap) {
		return
	}
	const dep = depsMap.get(key)
	if (dep) {
		dep.forEach(effect => effect())
	}
}

// 全局变量，存储当前激活的副作用函数
let activeEffect

// 定义副作用函数
function effect(fn) {
	const effectFn = () => {
		activeEffect = effectFn
		// 清理之前的依赖
		if (effectFn.deps) {
			effectFn.deps.forEach(dep => dep.delete(effectFn))
		}
		effectFn.deps = []
		const result = fn()
		activeEffect = null
		return result
	}
	effectFn()
	return effectFn
}

// 实现自定义的 computed 函数
function myComputed(getter) {
	let value
	let dirty = true // 标记是否需要重新计算

	const effectFn = effect(() => {
		dirty = true
		return getter()
	})

	const computedObject = {
		get value() {
			if (dirty) {
				value = effectFn()
				dirty = false
			}
			track(computedObject, 'value')
			return value
		}
	}

	return computedObject
}

// 实现自定义的`ref `函数（辅助实现响应式）
function myRef(initialValue) {
	const refObject = {
		get value() {
			track(refObject, 'value')
			return initialValue
		},
		set value(newValue) {
			if (initialValue !== newValue) {
				initialValue = newValue
				trigger(refObject, 'value')
			}
		}
	}
	return refObject
}

// 使用示例
const count = myRef(0)

const doubleCount = myComputed(() => {
	return count.value * 2
})

// 首次访问，进行计算
console.log(doubleCount.value)

// 修改依赖数据
count.value = 1

// 再次访问，因为依赖数据变化，重新计算
console.log(doubleCount.value)

    代码解释

    1. targetMap：是一个 WeakMap，用来存储对象及其依赖关系。每个对象对应一个 Map，Map 里每个键对应一个 Set，Set 中存储依赖该属性的副作用函数。
    2. track 函数：负责收集依赖。当读取响应式数据时，会调用该函数，把当前正在执行的副作用函数添加到依赖集合中。
    3. trigger 函数：用于触发依赖更新。当响应式数据发生变化时，会调用该函数，遍历依赖集合，执行其中的所有副作用函数。
    4. effect 函数：定义副作用函数，执行副作用函数前将其赋值给全局变量 activeEffect，方便在 track 函数中使用。同时会清理之前的依赖。
    5. myComputed 函数：
    • dirty 变量：标记是否需要重新计算，初始值为 true。
    • effectFn：是一个副作用函数，当依赖数据变化时，会将 dirty 置为 true，表示需要重新计算。
    • computedObject：返回一个包含 value 属性的对象，访问 value 属性时，如果 dirty 为 true，则重新计算并更新 value，同时将 dirty 置为 false。
    1. myRef 函数：辅助实现响应式数据，实现了简单的ref功能。
