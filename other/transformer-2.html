    遍历 torch.utils.data.DataLoader 有多种方式，具体取决于你的需求和代码风格。以下是几种常见的方法：

    # 方法一：直接迭代（最常用）

    for batch in train_loader:
  # batch 是一个包含批次数据的字典或元组
  input_ids = batch['input_ids']
  attention_mask = batch['attention_mask']
  labels = batch['labels']
  
  # 在这里进行训练步骤
  # model(input_ids, attention_mask=attention_mask, labels=labels)
  # ...

    # 方法二：使用 enumerate 获取批次索引

    for batch_idx, batch in enumerate(train_loader):
  input_ids = batch['input_ids']
  attention_mask = batch['attention_mask']
  labels = batch['labels']
  
  print(f"Processing batch {batch_idx}")
  # 训练代码...

    # 方法三：使用 tqdm 显示进度条

    from tqdm import tqdm

# 在训练循环中
for batch in tqdm(train_loader, desc="Training"):
  input_ids = batch['input_ids']
  attention_mask = batch['attention_mask']
  labels = batch['labels']
  
  # 训练代码...

    # 方法四：获取特定设备（GPU）

    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

for batch in train_loader:
  # 将数据移动到设备（GPU）
  input_ids = batch['input_ids'].to(device)
  attention_mask = batch['attention_mask'].to(device)
  labels = batch['labels'].to(device)
  
  # 训练代码...

    # 完整的训练循环示例

    from transformers import AdamW
from tqdm import tqdm

model = YourModel()  # 你的模型
optimizer = AdamW(model.parameters(), lr=5e-5)
model.train()

num_epochs = 3

for epoch in range(num_epochs):
print(f"Epoch {epoch + 1}/{num_epochs}")

# 使用进度条
for batch_idx, batch in enumerate(tqdm(train_loader, desc=f"Epoch {epoch+1}")):
  # 将数据移动到设备
  input_ids = batch['input_ids'].to(device)
  attention_mask = batch['attention_mask'].to(device)
  labels = batch['labels'].to(device)
  
  # 清零梯度
  optimizer.zero_grad()
  
  # 前向传播
  outputs = model(
    input_ids=input_ids,
    attention_mask=attention_mask,
    labels=labels
  )
  
  loss = outputs.loss
  
  # 反向传播
  loss.backward()
  
  # 更新参数
  optimizer.step()
  
  # 可选：打印训练信息
  if batch_idx % 100 == 0:
    print(f"Batch {batch_idx}, Loss: {loss.item():.4f}")

    # 方法五：使用 iter() 和 next() 手动控制

    # 创建迭代器
data_iter = iter(train_loader)

# 手动获取下一个批次
try:
  batch = next(data_iter)
  print(batch.keys())  # 查看批次中包含哪些键
except StopIteration:
  print("数据遍历完毕")

    # 调试技巧：查看批次结构

    # 首先获取一个批次看看结构
first_batch = next(iter(train_loader))
print("Batch type:", type(first_batch))
print("Batch keys:", first_batch.keys() if isinstance(first_batch, dict) else "Not a dict")
print("Input IDs shape:", first_batch['input_ids'].shape)
print("Attention mask shape:", first_batch['attention_mask'].shape)
print("Labels shape:", first_batch['labels'].shape)

    # 注意事项

    1. 批次大小：你设置了 batch_size=1，所以每个批次只有一个样本
    2. 数据格式：确保你知道 QADataSet 返回的数据格式（通常是字典）
    3. 设备移动：记得将数据移动到模型所在的设备（CPU/GPU）
    4. 梯度清零：在每个批次开始时调用 optimizer.zero_grad()
    5. 模型模式：训练时使用 model.train()，评估时使用 model.eval()

    推荐使用方法一或方法二，它们是最简洁和常用的方式。如果需要进度显示，可以结合方法三使用。
