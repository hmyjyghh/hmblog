import os
import json
import numpy as np
import torch
from torch.utils.data import Dataset, DataLoader
from datasets import Dataset as HFDataset
from transformers import AutoTokenizer, AutoModelForTokenClassification, TrainingArguments, Trainer, DataCollatorForTokenClassification, BitsAndBytesConfig
from peft import get_peft_model, LoraConfig, TaskType, PeftModel
from seqeval.metrics import classification_report, f1_score

categories = set()

# step2 定义继承自Dataset类的自定义类
class NERDataset(Dataset):
    def __init__(self, data_file):
        self.data = self.load_data(data_file)

    def load_data(self, data_file):
        Data = {}
        with open(data_file, 'rt', encoding='utf-8') as f:
            df = f.read().split('\n\n')
            for idx, line in enumerate(df):
                if not line:
                    break
                sentence, labels = '', []
                for i, item in enumerate(line.split('\n')):
                    char, tag = item.split(' ')
                    sentence += char
                    if tag.startswith('B'):
                        labels.append([i, i, char, tag[2:]]) # Remove the B- or I-
                        categories.add(tag[2:])
                    elif tag.startswith('I'):
                        labels[-1][1] = i
                        labels[-1][2] += char
                
                Data[idx] = {
                    'sentence': sentence,
                    'labels': labels, # 保留原始实体信息
                }
        print(f'数据集中包含的实体类型有：{categories}')
        return Data
    
    def __len__(self):
        return len(self.data)

    def __getitem__(self, idx):
        return self.data[idx]

# step3 加载数据集
train_data = NERDataset('./dataset/medical.train')
val_data = NERDataset('./dataset/medical.dev')

# 建立标签映射字典
id2label = {0:'O'}
for c in list(sorted(categories)):

    id2label[len(id2label)] = f"I-{c}"
label2id = {v: k for k, v in id2label.items()}
num_labels=len(id2label)

print(f'num_labels: {num_labels}')
print(label2id)

# 将数据集转换为 Dataset 对象
def to_hf_dataset(custom_dataset):
    sentences, labels = [], []
    for i in range(len(custom_dataset)):
        item = custom_dataset[i]
        sentences.append(item['sentence'])
        labels.append(json.dumps(item['labels'], ensure_ascii=False))
    
    return HFDataset.from_dict({
        'sentence': sentences,
        'labels': labels
    })

train_ds = to_hf_dataset(train_data)
eval_ds = to_hf_dataset(val_data)
test_ds = to_hf_dataset(test_data)

# 内存不足问题，使用QLoRA量化模型
bnb_config = BitsAndBytesConfig(

# step4 加载tokenizer 和 预训练模型
model_name = r'/root/autodl-tmp'
tokenizer = AutoTokenizer.from_pretrained(model_name, use_fast=True, trust_remote_code=True)

# 重要：添加pad_token

# step5 修正的数据预处理函数
def process_func(examples):
    sents = examples["sentence"]
    lb_strs = examples["labels"]
    
    batch_encoding = tokenizer(
        sents, 
        padding=False, 
        truncation=True, 
        max_length=512,
        return_offsets_mapping=True
    )
    
    all_labels = []
    
    for i, (sent, lb_str) in enumerate(zip(sents, lb_strs)):
        lbs = json.loads(lb_str)
        input_ids = batch_encoding["input_ids"][i]
        offset_mapping = batch_encoding["offset_mapping"][i]
        
        # 重要修正：初始化所有标签为-100（忽略的标签）
        labels = [-100] * len(input_ids)
        
        # 处理每个实体
        for char_start, char_end, entity_text, entity_type in lbs:
            entity_tokens = []
            
            # 找到覆盖该实体的所有token
            for token_idx, (token_start, token_end) in enumerate(offset_mapping):
                if token_start == token_end:  # 特殊token
                    continue
                
                # 检查token是否与实体有重叠
                if (token_start <= char_start and token_end >= char_end) or \
                   (token_start >= char_start and token_start < char_end) or \
                   (token_end > char_start and token_end <= char_end):
                    entity_tokens.append(token_idx)
            
            if not entity_tokens:
                print(f"警告: 无法为实体 '{entity_text}' 找到对应的token")
                continue
            
            # 分配B-I标签
            if entity_tokens:
                labels[entity_tokens[0]] = label2id[f"B-{entity_type}"]
                for token_idx in entity_tokens[1:]:
                    labels[token_idx] = label2id[f"I-{entity_type}"]
        
        # 处理非实体部分（标记为O）
        for j in range(len(labels)):
            if labels[j] == -100:  # 未被实体覆盖的token
                # 跳过特殊token（CLS, SEP, PAD等）
                if offset_mapping[j] != (0, 0):  # 非特殊token
                    labels[j] = label2id["O"]
        
        all_labels.append(labels)
    
    # 移除offset_mapping，不返回给模型
    batch_encoding.pop("offset_mapping")
    batch_encoding["labels"] = all_labels
    
    return batch_encoding

# 使用修正后的处理函数

def compute_metrics(eval_preds):
    logits, labels = eval_preds
    predictions = np.argmax(logits, axis=-1)
    
    true_labels = []
    true_predictions = []
    
    for i in range(len(labels)):
        preds = []
        trues = []
        for j in range(len(labels[i])):
            if labels[i][j] != -100:  # 忽略padding
                pred_label = id2label.get(predictions[i][j], "O")
                true_label = id2label.get(labels[i][j], "O")
                preds.append(pred_label)
                trues.append(true_label)
        
        true_predictions.append(preds)
        true_labels.append(trues)
    
    # 使用seqeval计算标准NER指标
    report = classification_report(true_labels, true_predictions, 
                                  output_dict=True, zero_division=0)
    
    # 提取每个实体类别的指标
    f1_scores = {}
    for entity_type in categories:
        b_label = f"B-{entity_type}"
        i_label = f"I-{entity_type}"
        
        entity_f1s = []
        if b_label in report:
            entity_f1s.append(report[b_label]['f1-score'])
        if i_label in report:
            entity_f1s.append(report[i_label]['f1-score'])
        
        if entity_f1s:
            f1_scores[entity_type] = np.mean(entity_f1s)
        else:
            f1_scores[entity_type] = 0.0
    
    # 总体指标
    macro_f1 = report['macro avg']['f1-score']
    micro_f1 = f1_score(true_labels, true_predictions, zero_division=0)
    
    result = {
        'eval_macro_f1': macro_f1,
        'eval_micro_f1': micro_f1,
        **{f'eval_{k}': v for k, v in f1_scores.items()}
    }
    
    return result
