    # Evaluate

    • evaluate 库是一个非常简单易用的机器学习模型评估函数库,
    • 只需要一行代码便可以加载各种任务的评估函数

    # 1. 加载评估函数

    import evaluate

accuracy = evaluate.load("accuracy")
precision = evaluate.load("precision")
recall = evaluate.load("recall")
f1 = evaluate.load("f1")
roc_auc = evaluate.load("roc_auc")
log_loss = evaluate.load("log_loss")
auc = evaluate.load("auc")
    1. accuracy(准确率)
    2. precision(精确率)
    3. recall(召回率)
    4. f1-score(f1 值)
    5. roc-auc(ROC 曲线下的面积)
    6. log-loss(对数损失)
    • 常用的 2 个参数
      1. predictions(预测值)
      2. reference(参考值)

    # 别人做的笔记

    Evaluate 基本用法：

    常见评估指标示例：

    • accuracy：准确率，计算预测正确的比例。
    • f1：F1 分数，精确率和召回率的调和平均。
    • precision：精确率。
    • recall：召回率。
    • bleu：用于机器翻译的 BLEU 分数。
    • rouge：用于摘要和生成的 ROUGE 分数。
    • meteor：METEOR 分数，用于翻译评估。
    • bertscore：基于 BERT 的语义相似度分数。
    • perplexity：困惑度，用于语言模型。
    • squad：用于问答的 SQuAD 评估。
    • glue：GLUE 基准的多种子指标（如 MRPC 的 F1 等）。

    # 评估指标汇总

    # 一、端到端评测（End-to-End Evaluation）的核心定义

    在Agent多轮对话系统中，端到端评测 是跳过中间模块（如意图识别、槽位填充、知识库检索等）的单独评测，直接从「用户输入（多轮对话上下文）」到「Agent最终回复」的全链路效果评估。核心是模拟真实用户使用场景，评估系统整体解决问题的能力，而非单个模块的性能。

    简单来说：

    • 模块化评测：只测「意图识别准不准」「槽位提取对不对」；
    • 端到端评测：测「用户说一段话（多轮），Agent能不能给出满足需求的回复」。

    # 二、端到端评测的核心指标

    常见的端到端评测指标分为两类：

    1. 自动指标：BLEU、ROUGE、METEOR（文本相似度）、Success Rate（任务完成率）、F1（回复正确性）等；
    2. 人工指标：回复相关性、流畅性、逻辑性、用户满意度（1-5分）等。

    其中，任务完成率（Success Rate） 是多轮对话Agent端到端评测的核心指标（比如：用户通过多轮对话完成「预订酒店」，是否真的完成了所有关键步骤）。

    # 三、Python代码示例：多轮对话Agent的端到端评测

    以下示例以「多轮酒店预订对话」为例，实现端到端评测的核心逻辑：

    1. 定义评测数据集（包含多轮上下文、预期完成的任务、预期回复）；
    2. 调用Agent系统获取实际回复；
    3. 计算「任务完成率」「回复相似度（BLEU）」等端到端指标。

    # 步骤1：安装依赖

    pip install nltk  # 用于计算BLEU/ROUGE
pip install sklearn  # 辅助指标计算

    # 步骤2：完整代码示例

    import nltk
from nltk.translate.bleu_score import sentence_bleu, SmoothingFunction
from typing import List, Dict, Tuple

# 下载nltk所需资源（首次运行需执行）
nltk.download('punkt')

# --------------------------
# 1. 定义端到端评测数据集（多轮对话场景）
# --------------------------
# 每条数据包含：多轮上下文、任务目标、预期回复、关键任务槽位
evaluation_dataset = [
    {
        "dialog_context": [
            "用户：我想订明天上海的酒店",
            "Agent：请问你需要什么价位的？",
            "用户：300-500元，靠近地铁口"
        ],
        "task_goal": "预订上海明天300-500元靠近地铁口的酒店",
        "expected_response": "为你推荐上海2号线南京东路站附近的XX酒店，价格420元/晚，符合你的300-500元预算，需要帮你预订吗？",
        "key_slots": {
            "city": "上海",
            "date": "明天",
            "price_range": "300-500元",
            "location": "靠近地铁口"
        }
    },
    {
        "dialog_context": [
            "用户：我要取消昨天订的北京酒店",
            "Agent：请问你的预订编号是多少？",
            "用户：NO123456"
        ],
        "task_goal": "取消编号NO123456的北京酒店预订",
        "expected_response": "已为你查询到预订编号NO123456的北京酒店订单，确认取消该订单吗？",
        "key_slots": {
            "action": "取消",
            "city": "北京",
            "order_id": "NO123456"
        }
    }
]

# --------------------------
# 2. 模拟Agent多轮对话回复函数（实际为调用Agent系统）
# --------------------------
def agent_response(dialog_context: List[str]) -> str:
    """
    模拟Agent根据多轮上下文生成回复
    实际场景中：这里会调用真实的Agent系统接口
    """
    last_user_utterance = dialog_context[-1].replace("用户：", "")
    if "上海" in last_user_utterance and "300-500" in last_user_utterance:
        return "为你推荐上海2号线南京东路站附近的XX酒店，价格420元/晚，符合你的300-500元预算，需要帮你预订吗？"
    elif "取消" in last_user_utterance and "NO123456" in last_user_utterance:
        return "已为你查询到预订编号NO123456的北京酒店订单，确认取消该订单吗？"
    else:
        return "抱歉，我没理解你的需求，请再说明一下。"

# --------------------------
# 3. 端到端评测核心逻辑
# --------------------------
class E2EConversationEvaluator:
    def __init__(self):
        self.smoother = SmoothingFunction()  # 解决BLEU分数为0的问题

    def calculate_bleu(self, pred_response: str, expected_response: str) -> float:
        """计算回复的BLEU分数（文本相似度）"""
        # 分词（中英文通用，nltk.punkt支持中文分句，分词需自定义，这里简化为按空格/字符拆分）
        pred_tokens = list(pred_response)  # 中文按字符拆分
        expected_tokens = list(expected_response)
        # 计算BLEU-4（n-gram=4）
        bleu = sentence_bleu(
            [expected_tokens], 
            pred_tokens, 
            smoothing_function=self.smoother.method4,
            weights=(0.25, 0.25, 0.25, 0.25)
        )
        return round(bleu, 4)

    def check_task_completion(self, pred_response: str, key_slots: Dict[str, str]) -> bool:
        """
        检查任务是否完成（核心端到端指标）：
        验证Agent回复中是否包含所有关键槽位（任务完成的核心信息）
        """
        for slot_key, slot_value in key_slots.items():
            if slot_value not in pred_response:
                return False
        return True

    def evaluate(self, dataset: List[Dict]) -> Dict[str, float]:
        """
        执行端到端评测，返回整体指标：
        - 任务完成率（Success Rate）
        - 平均BLEU分数（Average BLEU）
        """
        total_samples = len(dataset)
        completed_tasks = 0
        total_bleu = 0.0

        # 逐样本评测
        for idx, sample in enumerate(dataset):
            print(f"\n=== 评测样本 {idx+1} ===")
            # 1. 获取Agent实际回复
            pred_response = agent_response(sample["dialog_context"])
            # 2. 获取预期回复和关键槽位
            expected_response = sample["expected_response"]
            key_slots = sample["key_slots"]
            
            # 3. 计算指标
            bleu_score = self.calculate_bleu(pred_response, expected_response)
            task_completed = self.check_task_completion(pred_response, key_slots)
            
            # 4. 打印单样本结果
            print(f"上下文：{sample['dialog_context']}")
            print(f"预期回复：{expected_response}")
            print(f"Agent回复：{pred_response}")
            print(f"BLEU分数：{bleu_score}")
            print(f"任务完成：{'是' if task_completed else '否'}")

            # 累计指标
            total_bleu += bleu_score
            if task_completed:
                completed_tasks += 1

        # 计算整体指标
        success_rate = round(completed_tasks / total_samples * 100, 2)
        avg_bleu = round(total_bleu / total_samples, 4)

        # 返回整体评测结果
        overall_metrics = {
            "total_samples": total_samples,
            "completed_tasks": completed_tasks,
            "success_rate(%)": success_rate,
            "average_bleu": avg_bleu
        }
        return overall_metrics

# --------------------------
# 4. 执行评测并输出结果
# --------------------------
if __name__ == "__main__":
    evaluator = E2EConversationEvaluator()
    overall_metrics = evaluator.evaluate(evaluation_dataset)
    
    print("\n=== 端到端评测整体结果 ===")
    for metric_name, metric_value in overall_metrics.items():
        print(f"{metric_name}: {metric_value}")

    # 五、端到端评测的扩展说明

    1. 指标扩展：

      • 除了任务完成率和BLEU，还可加入「回复相关性评分（人工标注1-5分）」「多轮逻辑一致性」「错误回复率」等；
      • 对于开放域对话（非任务型），可加入「PPL（困惑度，衡量回复流畅性）」「多样性（distinct-1/distinct-2）」。

    2. 真实场景适配：

      • 代码中agent_response函数需替换为真实Agent的API调用；
      • 评测数据集需覆盖更多场景（边界case、多轮超长对话、歧义问题等）；
      • 可结合A/B测试，对比不同Agent版本的端到端效果。

    3. 与模块化评测的区别： 比如「意图识别模块评测」只看「用户说300-500元酒店」是否识别为「预订酒店意图」，而端到端评测不关心意图识别是否正确，只看最终是否能给出符合需求的回复（即使意图识别有小误差，但Agent通过多轮修正完成了任务，端到端评测仍判定为成功）。

    总结：端到端评测的核心是「以用户视角评估整体效果」，而非纠结单个模块的细节，这也是衡量Agent实际使用体验的最直接方式。
