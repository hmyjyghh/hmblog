    # 基于词的 f1 score

    def compute_metrics(eval_preds):
  """计算基于词的F1分数 - 适配您的标签体系"""
  logits, labels = eval_preds
  predictions = np.argmax(logits, axis=-1)
  
  # 移除padding和特殊token的标签
  true_labels = []
  true_predictions = []
  
  for i in range(len(labels)):
    preds = []
    trues = []
    for j in range(len(labels[i])):
      if labels[i][j] != -100:  # 忽略的标签
        preds.append(id2label[predictions[i][j]])
        trues.append(id2label[labels[i][j]])
    
    true_predictions.append(preds)
    true_labels.append(trues)
  
  # 计算每个实体类别的F1分数
  report = classification_report(true_labels, true_predictions, output_dict=True, zero_division=0)
  
  # 定义实体类别映射（将B-和I-标签合并为实体类别）
  entity_categories = {
      '中医治则': ['B-中医治则', 'I-中医治则'],
      '中医治疗': ['B-中医治疗', 'I-中医治疗'],
      '中医证候': ['B-中医证候', 'I-中医证候'],
      '中医诊断': ['B-中医诊断', 'I-中医诊断'],
      '中药': ['B-中药', 'I-中药'],
      '临床表现': ['B-临床表现', 'I-临床表现'],
      '其他治疗': ['B-其他治疗', 'I-其他治疗'],
      '方剂': ['B-方剂', 'I-方剂'],
      '西医治疗': ['B-西医治疗', 'I-西医治疗'],
      '西医诊断': ['B-西医诊断', 'I-西医诊断']
  }
  
  # 提取每个实体类别的F1分数
  f1_scores = {}
  for category, labels_in_category in entity_categories.items():
      category_f1_scores = []
      
      for label in labels_in_category:
          if label in report:
              category_f1_scores.append(report[label]['f1-score'])
      
      if category_f1_scores:
          # 取该类别的平均F1分数
          f1_scores[category] = np.mean(category_f1_scores)
      else:
          f1_scores[category] = 0.0
  
  # 计算宏平均F1和微平均F1
  macro_f1 = report['macro avg']['f1-score']
  micro_f1 = f1_score(true_labels, true_predictions, zero_division=0)
  
  result = {
      'macro_f1': macro_f1,
      'micro_f1': micro_f1,
      **f1_scores
  }
  
  return result

    def print_detailed_f1_scores(metrics):
    """打印每个实体类别的详细F1分数"""
    print("\n" + "="*60)
    print("各实体类别F1分数详细报告")
    print("="*60)
    
    # 提取实体类别的F1分数（排除总体指标）
    entity_scores = {}
    for key, value in metrics.items():
        if key not in ['eval_loss', 'eval_macro_f1', 'eval_micro_f1', 'eval_runtime', 
                      'eval_samples_per_second', 'eval_steps_per_second', 'epoch']:
            entity_scores[key] = value
    
    # 按F1分数排序
    sorted_scores = sorted(entity_scores.items(), key=lambda x: x[1], reverse=True)
    
    for entity, f1 in sorted_scores:
        print(f"{entity:>10}: {f1:.4f}")
    
    print("-"*60)
    if 'eval_macro_f1' in metrics:
        print(f"{'宏平均F1':>10}: {metrics['eval_macro_f1']:.4f}")
    if 'eval_micro_f1' in metrics:
        print(f"{'微平均F1':>10}: {metrics['eval_micro_f1']:.4f}")
    print("="*60)

    # 这个评估函数的特点：

    1. ✅ 基于词的F1score：使用seqeval库，它专门用于序列标注任务，计算的是实体级别的F1分数，不是单字级别的。

    2. ✅ 每种实体类别的F1score：

      • 将相关的B-标签和I-标签合并为实体类别
      • 例如：B-中药和I-中药合并为"中药"类别
      • 计算每个实体类别的平均F1分数

    3. ✅ 完整的实体类别覆盖：

      • 中医治则、中医治疗、中医证候、中医诊断
      • 中药、临床表现、其他治疗、方剂
      • 西医治疗、西医诊断

    4. ✅ 清晰的输出格式：通过print_detailed_f1_scores函数按分数排序打印所有实体类别的F1分数。
