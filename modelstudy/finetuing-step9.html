        # 获取tokenization结果
        encoding = tokenizer(sent, return_offsets_mapping=True, add_special_tokens=True)
        tokens = tokenizer.convert_ids_to_tokens(encoding["input_ids"])
        offsets = encoding["offset_mapping"]
        
        print("Token化结果:")
        for i, (token, offset) in enumerate(zip(tokens, offsets)):
            print(f"  {i:2d}: '{token:10}' -> 字符位置 {offset}")
        
        # 模拟标签分配
        labels = [label2id["O"]] * len(tokens)
        for char_start, char_end, entity_text, entity_type in entities:
            entity_tokens = []
            for token_idx, (token_start, token_end) in enumerate(offsets):
                if token_start == token_end:
                    continue
                if (token_start <= char_start < token_end) or (token_start < char_end <= token_end):
                    entity_tokens.append(token_idx)
            
            print(f"实体 '{entity_text}'(位置{char_start}-{char_end}) -> tokens: {entity_tokens}")
            
            if entity_tokens:
                labels[entity_tokens[0]] = label2id[f"B-{entity_type}"]
                for token_idx in entity_tokens[1:]:
                    labels[token_idx] = label2id[f"I-{entity_type}"]
        
        print("最终标签:")
        for i, (token, label) in enumerate(zip(tokens, labels)):
            label_name = id2label.get(label, "O")
            if label_name != "O":
                print(f"  {i:2d}: '{token:10}' -> {label_name}")

    # step8 创建评估函数
def compute_metrics(eval_preds):
  """
  计算评估指标
  """
  preds, labels = eval_preds
  preds = np.argmax(preds, axis=-1)

  # 移除填充部分
  true_labels = [[l for l in label if l != -100] for label in labels]
  true_preds = [
      [p for (p, l) in zip(pred, label) if l != -100]
      for (pred, label) in zip(preds, labels)
  ]

  # 将ID转换回标签
  true_labels = [tokenizer.decode(l, skip_special_tokens=True) for l in true_labels]
  true_preds = [tokenizer.decode(p, skip_special_tokens=True) for p in true_preds]

  # 计算准确率
  accuracy = sum(p == l for p, l in zip(true_preds, true_labels)) / len(true_labels)
  return {"accuracy": accuracy}

    # 简单计算token级别的准确率
