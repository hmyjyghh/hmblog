    Step 1: DataFrame 和 to_csv 函数详解

    寒梦

    # Step 1: DataFrame 和 to_csv 函数详解

    # DataFrame（数据帧）

    DataFrame 是pandas库中最核心的数据结构，相当于一个二维的表格数据。

    import pandas as pd

# 创建DataFrame的几种方式
# 1. 从字典创建
data = {
  '姓名': ['张三', '李四', '王五'],
  '年龄': [25, 30, 28],
  '城市': ['北京', '上海', '广州']
}
df = pd.DataFrame(data)
print(df)

# 2. 从列表创建
data_list = [
    ['张三', 25, '北京'],
    ['李四', 30, '上海'],
    ['王五', 28, '广州']
]
df2 = pd.DataFrame(data_list, columns=['姓名', '年龄', '城市'])

# DataFrame的基本属性
print(f"形状: {df.shape}")        # (3, 3)
print(f"列名: {df.columns.tolist()}")  # ['姓名', '年龄', '城市']
print(f"索引: {df.index.tolist()}")    # [0, 1, 2]

    # to_csv 函数

    to_csv() 用于将DataFrame保存为CSV文件：

    # 基本用法
df.to_csv('data.csv', index=False)

# 常用参数详解
df.to_csv(
    'data.csv',           # 文件名
    index=False,          # 不保存行索引
    header=True,          # 保存列名
    encoding='utf-8',     # 编码格式
    mode='w',             # 写入模式：'w'覆盖，'a'追加
    sep=',',              # 分隔符
    na_rep='NULL'         # 缺失值替换
)

# 在你的代码中的应用
def write(rows, header=False):
    df_new = pd.DataFrame(rows, index=range(len(rows)))
    df_new.to_csv(
        file_name, 
        mode='a',         # 追加模式
        header=header,    # 只有第一次写入表头
        index=False       # 不保存行号
    )

    # Step 2: 大模型应用开发中常用的pandas函数

    # 1. 数据读取函数

    # 读取各种格式数据
df = pd.read_csv('data.csv')           # CSV文件
df = pd.read_json('data.json')         # JSON文件
df = pd.read_excel('data.xlsx')        # Excel文件
df = pd.read_parquet('data.parquet')   # Parquet文件（高效）

    # 2. 数据探索函数

    # 基本信息查看
df.head(5)              # 前5行
df.info()               # 数据类型和内存信息
df.describe()           # 数值列统计信息
df.shape                # 数据形状
df.columns              # 列名

# 数据统计
df['score'].mean()      # 平均值
df['score'].max()       # 最大值
df['score'].value_counts()  # 值计数

    # 3. 数据清洗函数

    # 处理缺失值
df.dropna()             # 删除缺失值
df.fillna(0)            # 填充缺失值
df.isnull().sum()       # 统计每列缺失值数量

# 数据类型转换
df['timestamp'] = pd.to_datetime(df['timestamp'])
df['category'] = df['category'].astype('category')

# 去重和筛选
df.drop_duplicates()    # 去重
df[df['score'] > 0.8]  # 条件筛选

    # 4. 数据处理函数

    # 数据分组聚合（重要！）
df.groupby('category')['score'].mean()  # 按类别分组求平均分
df.groupby('user_id').agg({
    'score': ['mean', 'max', 'count'],
    'text': 'count'
})

# 数据排序
df.sort_values('score', ascending=False)  # 按分数降序排列

# 数据合并
pd.concat([df1, df2])                    # 纵向合并
pd.merge(df1, df2, on='user_id')         # 横向合并

    # 5. 文本数据处理（大模型应用重点）

    # 文本列操作
df['text_length'] = df['text'].str.len()           # 文本长度
df['clean_text'] = df['text'].str.lower()          # 转小写
df['word_count'] = df['text'].str.split().str.len() # 词数统计
df['contains_keyword'] = df['text'].str.contains('AI')  # 包含关键词

# 应用自定义函数（常用于大模型处理）
df['embedding'] = df['text'].apply(get_embedding)  # 生成嵌入向量
df['sentiment'] = df['text'].apply(analyze_sentiment)  # 情感分析

    # 6. 数据集划分

    # 训练集测试集划分
from sklearn.model_selection import train_test_split
train_df, test_df = train_test_split(df, test_size=0.2, random_state=42)

# 按条件划分
train_df = df[df['split'] == 'train']
test_df = df[df['split'] == 'test']

    # 7. 批处理函数（处理大量数据）

    # 分批处理大数据集
batch_size = 1000
for i in range(0, len(df), batch_size):
  batch_df = df.iloc[i:i+batch_size]
  process_batch(batch_df)  # 处理每个批次

# 使用迭代器
for chunk in pd.read_csv('large_file.csv', chunksize=1000):
  process_chunk(chunk)

    # 8. 特征工程函数

    # 创建时间特征
df['hour'] = df['timestamp'].dt.hour
df['day_of_week'] = df['timestamp'].dt.dayofweek

# 独热编码
df_encoded = pd.get_dummies(df, columns=['category'])

# 数据标准化
from sklearn.preprocessing import StandardScaler
df['score_normalized'] = StandardScaler().fit_transform(df[['score']])
