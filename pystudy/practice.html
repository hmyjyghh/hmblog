            sentences = content.split('\n\n')
            
            for sent in sentences:
                sentence = []
                labels = []
                # 按行分割每个字符和标签
                for line in sent.split('\n'):
                    # 跳过空行或格式不正确的行
                    if not line.strip() or ' ' not in line:
                        continue
                    # 分割字符和标签
                    char, tag = line.split(' ', 1)  # 限制只分割一次，防止标签中包含空格
                    sentence.append(char)
                    labels.append(tag)
                
                if sentence:  # 只添加非空句子
                    data.append((sentence, labels))
        
        return data

    def load_data(self, data_file):
        Data = {}
        categories = set()
        with open(data_file, 'rt', encoding='utf-8') as f:
            # print('start------', f.read().split('\n\n'))
            df = f.read().split('\n\n')
            for idx, line in enumerate(df):
                # print('=' * 10, line)
                if not line:
                    break
                sentence, labels = '', [] 
                for i, item in enumerate(line.split('\n')):
                    # print(item, '------------') 海 O
                    char, tag = item.split(' ')
                    sentence += char
                    # print(char, tag)
                    if tag.startswith('B'):
                        print(char, tag, 'B-开始', i, i)
                        labels.append([i, i, char, tag[2:]]) # Remove the B- or I-
                        categories.add(tag[2:])
                    elif tag.startswith('I'):
                        # i = 8, 门  I-LOC
                        print(char, tag, 'I-中间') 
                        # labels = [[7, 7, '厦', 'LOC']]
                        labels[-1][1] = i
                        labels[-1][2] += char
                Data[idx] = {
                    'sentence': sentence, 
                    'labels': labels
                }
            print(Data)
            print(f'数据集中包含的实体类型有：{categories}')
        return Data
