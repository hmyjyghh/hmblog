    # zip()

    用于创建一个聚合多个可迭代对象的迭代器。

    • zip()函数用于创建一个聚合多个可迭代对象的迭代器。
    • 它会将作参数传入的每个可迭代对象的每个元素依次组合成元组，即第i个元组包含来自每个参数的第i个元素。
    x = [1, 2, 3]
y = [4, 5, 6]

zipped = zip(x, y)

print(zipped) # <zip object at 0x0000023340FFE440>
print(list(zipped)) # [(1, 4), (2, 5), (3, 6)]

x = [1, 2, 3]
y = [4, 5, 6]
# z = [7, 8, 9]

# or
z = "Hmyjy"

zipped = zip(x, y, z)

print(zipped) # <zip object at 0x0000023340FFE440>
print(list(zipped)) # [(1, 4, 'H'), (2, 5, 'm'), (3, 6, 'y')] # 少了后面2位: jy

# 也可以使用 itertools

import itertools

x = [1, 2, 3]
y = [4, 5, 6]
z = "Hmyjy"

zipped = itertools.zip_longest(x, y, z)
print(list(zipped)) # [(1, 4, 'H'), (2, 5, 'm'), (3, 6, 'y'), (None, None, 'j'), (None, None, 'y')] # 没有的值，补：None

    # enumerate()

    enumerate() 函数用于返回一个枚举对象，它的功能就是将可迭代对象中的每个元素及从0开始的序号共同构成一个二元组的列表。

    • 是用于返回一个枚举对象
    seasons = ["Spring", "Summer", "Fall", "Winter"]

print(enumerate(seasons)) # <enumerate object at 0x0000022EFC68ADE0>

print(list(enumerate(seasons))) # [(0, 'Spring'), (1, 'Summer'), (2, 'Fall'), (3, 'Winter')]


seasons = ["Spring", "Summer", "Fall", "Winter"]

print(list(enumerate(seasons, 10))) # 指定序号，从10 开始
# [(10, 'Spring'), (11, 'Summer'), (12, 'Fall'), (13, 'Winter')]

    # enumerate 和 zip 结合使用， 一般是为了取索引

    # 构造 documents 数据：包含6个模拟的文本文档（可以是任意字符串，模拟实际业务中的文档内容）
documents = [
    "人工智能在医疗领域的应用与发展",
    "机器学习算法优化与模型训练技巧",
    "深度学习框架TensorFlow的使用指南",
    "春季花卉种植方法与病虫害防治",
    "阳台盆栽蔬菜的养护技巧分享",
    "园林植物修剪与日常管理要点"
]

# 构造 labels 数据：对应6个文档的聚类标签（聚类标签通常为整数，这里分为2个聚类：0和1）
# 规律：前3个是人工智能/机器学习相关（聚类0），后3个是植物种植相关（聚类1）
labels = [0, 0, 0, 1, 1, 1]

# print(zip(documents, labels)) # <zip object at 0x000001ED12657C00>

for i, (doc, label) in enumerate(zip(documents, labels)):
    print(f"文档{i+1}: '{doc}' -> 聚类{label}")

# 文档1: '人工智能在医疗领域的应用与发展' -> 聚类0
# 文档2: '机器学习算法优化与模型训练技巧' -> 聚类0
# 文档3: '深度学习框架TensorFlow的使用指南' -> 聚类0
# 文档4: '春季花卉种植方法与病虫害防治' -> 聚类1
# 文档5: '阳台盆栽蔬菜的养护技巧分享' -> 聚类1
# 文档6: '园林植物修剪与日常管理要点' -> 聚类1

    # all() & any()

    是否存在某个元素的值，为真

    x = [1, 1, 0]
y = [1, 1, 9]

print(all(x), 'x') # False

print(all(y), 'y') # True

print(any(x)) # True
print(any(y)) # True
